# Skill: nestjs-domain-scaffold

Use this skill when scaffolding a new domain module. It provides the canonical directory layout, file templates, and a documentation update checklist.

---

## Directory Layout

```
src/<domain>/
├── <domain>.module.ts
├── controllers/
│   └── <domain>.controller.ts
├── services/
│   └── <domain>.service.ts          # or split: command / query / domain
├── dto/
│   ├── create-<domain>.dto.ts
│   ├── update-<domain>.dto.ts
│   ├── query-<domain>.dto.ts
│   ├── <domain>-response.dto.ts
│   └── <domain>-paginated-response.dto.ts
└── entities/
    └── <domain>.entity.ts
```

Register the module in `AppModule`.

## Entity Template

```ts
import { Entity, Column, PrimaryColumn } from 'typeorm';
import { v4 as uuidv4 } from 'uuid';

@Entity('<domain>s')
export class DomainEntity {
  @PrimaryColumn('uuid')
  id: string;

  @Column({ type: 'varchar', length: 255 })
  name: string;

  @Column({ type: 'boolean', default: true })
  isActive: boolean;

  @Column({ type: 'bigint' })
  createdAt: number;

  @Column({ type: 'bigint' })
  updatedAt: number;

  @Column({ type: 'bigint', nullable: true, default: null })
  deletedAt: number | null;

  constructor() {
    this.id = uuidv4();
  }
}
```

Key rules:
- `@PrimaryColumn('uuid')` — UUID assigned in constructor, not auto-generated by DB.
- Timestamps as Unix milliseconds (`bigint` column, `number` in TypeScript).
- `deletedAt: number | null` for soft deletes. Never `@DeleteDateColumn`.
- Explicit column types always. No `@CreateDateColumn`/`@UpdateDateColumn`.

## Controller Template

```ts
import { Controller, Get, Post, Patch, Delete, Body, Param, Query, UseGuards } from '@nestjs/common';
import { ApiTags, ApiOperation, ApiResponse, ApiBadRequestResponse } from '@nestjs/swagger';
import { ApiKeyGuard } from '../../common/guards/api-key.guard';
import { ErrorResponseDto } from '../../common/error/error-response.dto';
import { DomainService } from '../services/domain.service';
import { CreateDomainDto } from '../dto/create-domain.dto';
import { UpdateDomainDto } from '../dto/update-domain.dto';
import { QueryDomainDto } from '../dto/query-domain.dto';
import { DomainResponseDto } from '../dto/domain-response.dto';
import { DomainPaginatedResponseDto } from '../dto/domain-paginated-response.dto';

@ApiTags('Domain')
@ApiBadRequestResponse({ type: ErrorResponseDto })
@Controller('domain')
@UseGuards(ApiKeyGuard)
export class DomainController {
  constructor(private readonly domainService: DomainService) {}

  @Get()
  @ApiOperation({ summary: 'List with pagination' })
  @ApiResponse({ status: 200, type: DomainPaginatedResponseDto })
  findAll(@Query() query: QueryDomainDto) {
    return this.domainService.findAll(query);
  }

  @Get(':id')
  @ApiOperation({ summary: 'Get by ID' })
  @ApiResponse({ status: 200, type: DomainResponseDto })
  @ApiResponse({ status: 404, type: ErrorResponseDto })
  findOne(@Param('id') id: string) {
    return this.domainService.findOne(id);
  }

  @Post()
  @ApiOperation({ summary: 'Create' })
  @ApiResponse({ status: 201, type: DomainResponseDto })
  create(@Body() dto: CreateDomainDto) {
    return this.domainService.create(dto);
  }

  @Patch(':id')
  @ApiOperation({ summary: 'Update' })
  @ApiResponse({ status: 200, type: DomainResponseDto })
  @ApiResponse({ status: 404, type: ErrorResponseDto })
  update(@Param('id') id: string, @Body() dto: UpdateDomainDto) {
    return this.domainService.update(id, dto);
  }

  @Delete(':id')
  @ApiOperation({ summary: 'Delete' })
  @ApiResponse({ status: 200, type: DomainResponseDto })
  @ApiResponse({ status: 404, type: ErrorResponseDto })
  remove(@Param('id') id: string) {
    return this.domainService.remove(id);
  }
}
```

## DTO Templates

**Create DTO:**

```ts
import { IsString, IsNotEmpty, IsOptional, IsBoolean, MaxLength } from 'class-validator';
import { Transform } from 'class-transformer';
import { ApiProperty } from '@nestjs/swagger';

export class CreateDomainDto {
  @ApiProperty({ description: 'Name', example: 'Widget', maxLength: 255 })
  @IsString()
  @IsNotEmpty()
  @MaxLength(255)
  @Transform(({ value }) => value?.trim())
  name: string;

  @ApiProperty({ description: 'Active status', example: true, required: false, default: true })
  @IsOptional()
  @IsBoolean()
  isActive?: boolean;
}
```

**Query DTO:**

```ts
import { IsInt, IsOptional, Min, IsString } from 'class-validator';
import { Type } from 'class-transformer';
import { ApiProperty } from '@nestjs/swagger';

export class QueryDomainDto {
  @ApiProperty({ description: 'Search query', example: 'widget', required: false })
  @IsOptional()
  @IsString()
  q?: string;

  @ApiProperty({ description: 'Page number', example: 1, required: false, minimum: 1, default: 1 })
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  page?: number = 1;

  @ApiProperty({ description: 'Items per page', example: 20, required: false, minimum: 1, default: 20 })
  @IsOptional()
  @Type(() => Number)
  @IsInt()
  @Min(1)
  limit?: number = 20;
}
```

## Service Template

```ts
import { Injectable, NotFoundException, BadRequestException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, IsNull } from 'typeorm';
import { validate as uuidValidate } from 'uuid';
import { DomainEntity } from '../entities/domain.entity';
import { CreateDomainDto } from '../dto/create-domain.dto';
import { UpdateDomainDto } from '../dto/update-domain.dto';
import { QueryDomainDto } from '../dto/query-domain.dto';
import { LoggingService } from '../../common/logging/logging.service';

@Injectable()
export class DomainService {
  constructor(
    @InjectRepository(DomainEntity)
    private readonly repo: Repository<DomainEntity>,
    private readonly logger: LoggingService,
  ) {}

  async findAll(query: QueryDomainDto) {
    const { page = 1, limit = 20, q } = query;
    const skip = (page - 1) * limit;

    const qb = this.repo.createQueryBuilder('domain');
    qb.where('domain.deletedAt IS NULL');

    if (q) {
      qb.andWhere('domain.name ILIKE :q', { q: `%${q}%` });
    }

    const [data, total] = await qb
      .orderBy('domain.createdAt', 'DESC')
      .skip(skip)
      .take(limit)
      .getManyAndCount();

    return {
      data,
      meta: { total, page, limit, totalPages: Math.ceil(total / limit) },
    };
  }

  async findOne(id: string): Promise<DomainEntity> {
    if (!uuidValidate(id)) {
      throw new BadRequestException(`Invalid ID format: ${id}`);
    }

    const entity = await this.repo.findOne({ where: { id, deletedAt: IsNull() } });

    if (!entity) {
      throw new NotFoundException(`Domain item with ID ${id} not found`);
    }

    return entity;
  }

  async create(dto: CreateDomainDto): Promise<DomainEntity> {
    const now = Date.now();
    const entity = this.repo.create({
      name: dto.name,
      isActive: dto.isActive ?? true,
      createdAt: now,
      updatedAt: now,
      deletedAt: null,
    });

    const saved = await this.repo.save(entity);
    this.logger.info('Domain item created', { id: saved.id });
    return saved;
  }

  async update(id: string, dto: UpdateDomainDto): Promise<DomainEntity> {
    const entity = await this.findOne(id);

    if (dto.name !== undefined) entity.name = dto.name;
    if (dto.isActive !== undefined) entity.isActive = dto.isActive;
    entity.updatedAt = Date.now();

    const updated = await this.repo.save(entity);
    this.logger.info('Domain item updated', { id: updated.id });
    return updated;
  }

  async remove(id: string): Promise<DomainEntity> {
    const entity = await this.findOne(id);
    entity.deletedAt = Date.now();
    entity.updatedAt = Date.now();

    const removed = await this.repo.save(entity);
    this.logger.info('Domain item deleted', { id: removed.id });
    return removed;
  }
}
```

Key conventions:
- UUID validation via `validate` from the `uuid` package (already a project dependency).
- Always filter `deletedAt IS NULL` — use `IsNull()` from TypeORM for typed `findOne`, raw `IS NULL` in `createQueryBuilder`.
- `create` initializes both `createdAt` and `updatedAt`. `update` and `remove` set only `updatedAt`.
- Return the entity from `create`, `update`, and `remove` — controllers rely on this.
- `logger.info` after every write, `logger.error` on unexpected failures.

## Module Template

```ts
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { DomainEntity } from './entities/domain.entity';
import { DomainController } from './controllers/domain.controller';
import { DomainService } from './services/domain.service';
import { LoggingService } from '../common/logging/logging.service';

@Module({
  imports: [TypeOrmModule.forFeature([DomainEntity])],
  controllers: [DomainController],
  providers: [
    DomainService,
    { provide: LoggingService, useFactory: () => new LoggingService('DomainService') },
  ],
})
export class DomainModule {}
```

## Post-Scaffold Checklist

- [ ] Module registered in `AppModule`
- [ ] Entity file created in `src/<domain>/entities/`
- [ ] Migration generated: `npm run migration:generate -- src/migrations/CreateDomainTable`
- [ ] Service spec file created (see `minimal-test-strategy` skill)
- [ ] `README.md` updated (add domain to structure and feature list)
- [ ] `.env.example` updated if new env variables are needed
